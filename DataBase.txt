create Database Consultorio

use Consultorio

create table armazon (
armazonid int not null auto_increment primary key,
marca varchar(50),
modelo varchar(50),
color varchar(50),
tipo_de_lente varchar(50),
material varchar(50),
precio float,
cantidad_disponible integer
);


create table mica(
micaid int not null auto_increment primary key,
nombre varchar(50),
descripcion varchar(50),
precio float,
cantidad_disponible integer
);

create table usuario(
usuarioid int not null auto_increment primary key,
nombre varchar(50),
aPaterno varchar(50),
aMaterno varchar(50),
nombreUsuario varchar(50),
telefono varchar(50),
correo varchar(50),
passwordHasH BLOB,
passwordSalt BLOB
);
